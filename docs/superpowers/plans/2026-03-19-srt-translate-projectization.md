```ts
  timecode: string;
  text: string;
  translated: string | null;
  status: SubtitleStatus;
```

```ts
import type { SubtitleEntry } from './types';

export function parseSrt(content: string): SubtitleEntry[] {
  const normalized = content.replace(/\r\n/g, '\n').replace(/\r/g, '\n').trim();
  if (!normalized) return [];

  return normalized.split(/\n{2,}/).flatMap((block, index) => {
    const lines = block.trim().split('\n');
    if (lines.length < 3 || !lines[1]?.includes('-->')) return [];

    const parsedIndex = Number.parseInt(lines[0] ?? '', 10);

    return [
      {
        idx: Number.isNaN(parsedIndex) ? index + 1 : parsedIndex,
        timecode: lines[1].trim(),
        text: lines.slice(2).join('\n').replace(/<[^>]+>/g, '').trim(),
        translated: null,
        status: 'pending',
      },
    ].filter((entry) => entry.text.length > 0);
  });
}

export function serializeSrt(entries: SubtitleEntry[]): string {
  return `${entries
    .map((entry, index) => `${index + 1}\n${entry.timecode}\n${entry.translated || entry.text}`)
    .join('\n\n')}\n`;
}
```

```ts
export function buildTranslationMessages(texts: string[], contextTexts: string[]) {
  const system = '你是专业的日语字幕翻译员...必须严格返回JSON数组格式。';
  const contextBlock =
    contextTexts.length > 0
      ? `\n【前文参考】\n${contextTexts.map((text, index) => `${index + 1}. ${text}`).join('\n')}\n`
      : '';

  return {
    system,
    user: `${contextBlock}\n【待翻译字幕】\n${texts.map((text, index) => `${index + 1}. ${text}`).join('\n')}`,
  };
}
```

```ts
export function parseTranslationResponse(text: string, count: number): string[] {
  const stripped = text.replace(/^```(?:json)?\n?/m, '').replace(/\n?```$/m, '').trim();

  try {
    const parsed = JSON.parse(stripped);
    if (Array.isArray(parsed) && parsed.length > 0) {
      return parsed.slice(0, count).map(String);
    }
  } catch {}

  const lines = text
    .split('\n')
    .map((line) => line.replace(/^\d+[.。、:：]\s*/, '').trim())
    .filter(Boolean);

  if (lines.length >= count) {
    return lines.slice(0, count);
  }

  return Array(count).fill('[翻译失败]');
}
```

```ts
export async function translateWithOpenAiCompatible(
  texts: string[],
  contextTexts: string[],
  config: Record<string, string>,
  signal: AbortSignal,
) {
  const endpoint = (config.endpoint || 'https://api.openai.com/v1').replace(/\/$/, '');
  const { system, user } = buildTranslationMessages(texts, contextTexts);

  const response = await fetch(`${endpoint}/chat/completions`, {
    method: 'POST',
    signal,
    headers: {
      'Content-Type': 'application/json',
      Authorization: `Bearer ${config.apiKey}`,
    },
    body: JSON.stringify({
      model: config.model || 'gpt-4o-mini',
      temperature: Number.parseFloat(config.temperature || '0.3') || 0.3,
      messages: [
        { role: 'system', content: system },
        { role: 'user', content: user },
      ],
    }),
  });

  if (!response.ok) throw new Error(`API ${response.status}`);

  const data = await response.json();
  return parseTranslationResponse(data.choices[0].message.content, texts.length);
}
```

```ts
export async function runTranslation(entries: SubtitleEntry[], options: RunTranslationOptions) {
  const results = entries.map((entry) => ({ ...entry }));

  for (let index = 0; index < entries.length; index += options.batchSize) {
    const end = Math.min(index + options.batchSize, entries.length);
    const batch = entries.slice(index, end);
    const context = results
      .slice(Math.max(0, index - options.contextLines), index)
      .map((entry) => entry.translated)
      .filter(Boolean) as string[];

    try {
      const translated = await options.dispatchTranslate(batch.map((entry) => entry.text), context);
      translated.forEach((text, offset) => {
        results[index + offset] = {
          ...results[index + offset],
          translated: text,
          status: text && text !== '[翻译失败]' ? 'done' : 'error',
        };
      });
    } catch (error) {
      batch.forEach((_, offset) => {
        results[index + offset] = { ...results[index + offset], status: 'error' };
      });
      options.onLog(String(error));
    }

    options.onUpdate([...results], (end / entries.length) * 100);
  }

  return results;
}
```
